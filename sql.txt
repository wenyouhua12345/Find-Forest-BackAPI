    role character varying(50) COLLATE pg_catalog."default",
    employee_id character varying(50) COLLATE pg_catalog."default",
    department character varying(50) COLLATE pg_catalog."default",
    phone character varying(50) COLLATE pg_catalog."default",
    email character varying(50) COLLATE pg_catalog."default",
    avatar_url character varying(255) COLLATE pg_catalog."default",
    updated_at character varying(50) COLLATE pg_catalog."default",
    CONSTRAINT "logins-user" FOREIGN KEY (id)

        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID
